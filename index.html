def extract_media(entities):
    if isinstance(entities, dict) and 'media' in entities:
        if isinstance(entities, dict) and 'type' in entities['media'][0]:
            return entities['media'][0]['type']
    else:
        return None

    if isinstance(entities, dict) and 'urls' in entities and len(entities['urls'])>0:
        if 'url' in entities['urls'][0]:
            return entities['urls'][0]['url']
    else:
        return None

def get_hashtag(data):
    hashtag_list = []
    for i in data.iterrows():
        pack = i[1]['entities']['hashtags'][:] if len(i[1]['entities']['hashtags'])!=0 else None
        draft = []
        if pack is not None:
                if len(pack) == 1:
                    draft.append([pack[0]['text']])
                else:
                    list_draft = []
                    list_draft.append(pack[0]['text'])
                    for i in range(1,len(pack)):
                        list_draft.append(pack[i]['text'])
                    draft.append(list_draft)
        else:
            draft.append([])
        hashtag_list.extend(draft)
    return hashtag_list

def process_text(data_text):
    url_pattern = r'https?://\S+|www\.\S+'
    pattern = r'([^A-Za-z\d#@\'"]|\'(?![A-Za-z\d#@]))'
    process_text = data_text.apply(lambda x: re.sub(url_pattern, '', x))
    process_text = process_text.apply(lambda x: re.split(pattern, x))
    process_text = process_text.apply(lambda words: [word for word in words if re.search(r'[A-Za-z]', word)])
    return process_text

data['process_text'] = process_text(data['full_text'])

def remove_stop_words(data, name=data['user']):
    data_word=[]
    for i in data[data['user']==name]['process_text']:
        data_word.extend(i)
    filtered_word = [w.lower() for w in data_word if not w.lower() in stop_words]
    frequency_data = collections.Counter(filtered_word)
    return frequency_data, filtered_word

def plot_frequency(frequency_data, filtered_data_word, name='Data', ax=None):
    list_log_word = {}
    for i in frequency_data.items():
        log_word = math.log((i[1]+1)/(len(filtered_data_word)+1))
        list_log_word[i[0]]=log_word

    list_log_word= sorted(list_log_word.items(), key=lambda x: x[1], reverse=True)

    if ax == None:
        fig, ax = plt.subplots(figsize=(10, 8))
        word = [i[0] for i in list_log_word[:20]]
        logodd = [i[1] for i in list_log_word[:20]]
        sns.barplot(x=logodd, y=word, data=data, ax=ax)
        plt.xlabel('Log Odds Ratio')
        plt.ylabel('Words')
        plt.title('Top 20 words have highest log ratio of '+ name)
        plt.show()
    else:
        word = [i[0] for i in list_log_word[:20]]
        logodd = [i[1] for i in list_log_word[:20]]
        sns.barplot(x=logodd, y=word, data=data, ax=ax)
        ax.set_xlabel('Log Odds Ratio')
        ax.set_ylabel('Words')
        ax.set_title('Top 20 words have highest log ratio of '+ name)
    return list_log_word
